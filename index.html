CombatManager.lua (Server)

-- CombatManager is the only script that touches damage logic.
-- All other systems go through it via ModuleScript calls, never directly.

local CombatManager = {}

local HitboxService  = require(ServerModules.HitboxService)
local IFrameService  = require(ServerModules.IFrameService)
local StunService    = require(ServerModules.StunService)
local RagdollService = require(ServerModules.RagdollService)

local MAX_REACH      = 9
local LATENCY_BUFFER = 0.35
local RATE_WINDOW    = 0.4

local hitTimestamps  = {}

local function isRateLimited(player)
  local now  = os.clock()
  local last = hitTimestamps[player] or 0
  if now - last < RATE_WINDOW then return true end
  hitTimestamps[player] = now
  return false
end

local function validateHit(attacker, hitData)
  local root = attacker.Character?.HumanoidRootPart
  if not root then return false end

  local dist = (root.Position - hitData.origin).Magnitude
  if dist > MAX_REACH then return false end

  if os.clock() - hitData.timestamp > LATENCY_BUFFER then return false end

  return true
end

function CombatManager.ProcessHit(attacker, target, hitData)
  if isRateLimited(attacker) then return end
  if not validateHit(attacker, hitData) then return end
  if IFrameService.IsInvincible(target) then return end

  local dmg = hitData.damage
  target.Character.Humanoid.Health -= dmg

  StunService.Apply(target, hitData.stunDuration)

  if target.Character.Humanoid.Health <= 0 then
    RagdollService.Apply(target, 3)
  end
end

return CombatManager
